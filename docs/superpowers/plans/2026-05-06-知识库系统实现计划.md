**Goal:** 搭建完整知识库系统，支持采集数据接入、文档上传、人工录入、语义检索，为 AI 问答提供知识支撑

**Architecture:**
- Python FastAPI 服务 (端口 5002) 处理知识库核心逻辑

- Java 后端在采集完成后触发知识库接入

**Tech Stack:** Python 3.10+, FastAPI, BGE, Qdrant, pdfplumber, python-docx, MySQL, MinIO, Element Plus Vue

```python
from contextlib import contextmanager
import os

DB_CONFIG = {
    "host": os.getenv("MYSQL_HOST", "localhost"),
    "port": int(os.getenv("MYSQL_PORT", "3306")),
    "user": os.getenv("MYSQL_USER", "root"),
    "password": os.getenv("MYSQL_PASSWORD", "password"),
    "database": os.getenv("MYSQL_DATABASE", "scfx"),
    "charset": "utf8mb4"
}

@contextmanager
def get_connection():
    conn = pymysql.connect(**DB_CONFIG)
    try:
        yield conn
    finally:
        conn.close()

def execute_query(sql, params=None):
    with get_connection() as conn:
        with conn.cursor(pymysql.cursors.DictCursor) as cursor:
            cursor.execute(sql, params)
            return cursor.fetchall()

def execute_update(sql, params=None):
    with get_connection() as conn:
        with conn.cursor() as cursor:
            cursor.execute(sql, params)
            conn.commit()
            return cursor.lastrowid
```

```python
import re

def chunk_text(text: str, min_chunk_size: int = 500, max_chunk_size: int = 800, overlap: int = 50) -> list:
    """
    将文本分块，每块 500-800 字，块之间保留重叠以保持上下文
    """
    # 按段落分割
    paragraphs = re.split(r'\n\s*\n', text)
    paragraphs = [p.strip() for p in paragraphs if p.strip()]

    chunks = []
    current_chunk = ""
    current_size = 0

    for para in paragraphs:
        para_len = len(para)
        if current_size + para_len <= max_chunk_size:
            current_chunk += para + "\n\n"
            current_size += para_len + 2
        else:
            # 当前块已满
            if current_chunk:
                chunks.append(current_chunk.strip())
            # 新块开始，保留重叠
            if overlap > 0 and len(current_chunk) > overlap:
                overlap_text = current_chunk[-overlap:]
                current_chunk = overlap_text + para + "\n\n"
            else:
                current_chunk = para + "\n\n"
            current_size = len(para)

    if current_chunk.strip():
        chunks.append(current_chunk.strip())

    # 如果仍然太大（没有段落边界），直接按长度分割
    final_chunks = []
    for chunk in chunks:
        if len(chunk) > max_chunk_size:
            for i in range(0, len(chunk), max_chunk_size - overlap):
                part = chunk[i:i + max_chunk_size]
                if part.strip():
                    final_chunks.append(part.strip())
        else:
            final_chunks.append(chunk)

    return final_chunks
```

```python
import os
import tempfile
import pdfplumber
from docx import Document
from typing import Optional

def parse_pdf(file_path: str) -> str:
    """从 PDF 提取文本"""
    text_parts = []
    with pdfplumber.open(file_path) as pdf:
        for page in pdf.pages:
            page_text = page.extract_text()
            if page_text:
                text_parts.append(page_text)
    return "\n\n".join(text_parts)

def parse_docx(file_path: str) -> str:
    """从 Word 文档提取文本"""
    doc = Document(file_path)
    paragraphs = [p.text for p in doc.paragraphs if p.text.strip()]
    return "\n\n".join(paragraphs)

def parse_txt(file_path: str) -> str:
    """读取 TXT 文件"""
    with open(file_path, 'r', encoding='utf-8') as f:
        return f.read()

def parse_document(file_path: str, file_type: str) -> str:
    """根据文件类型解析"""
    if file_type == 'pdf':
        return parse_pdf(file_path)
    elif file_type in ['doc', 'docx']:
        return parse_docx(file_path)
    elif file_type == 'txt':
        return parse_txt(file_path)
    else:
        raise ValueError(f"Unsupported file type: {file_type}")

def clean_text(text: str) -> str:
    """清洗文本，去除特殊字符"""
    import re
    # 去除多余空白
    text = re.sub(r'\s+', ' ', text)
    # 去除特殊控制字符
    text = re.sub(r'[\x00-\x08\x0b-\x0c\x0e-\x1f\x7f]', '', text)
    return text.strip()
```

```python
def embed_text(text: str) -> list:
    """使用 BGE 将文本转为向量"""
    model = get_model()
    embedding = model.encode(text, normalize=True)
    return embedding.tolist()

def embed_batch(texts: list) -> list:
    """批量向量化"""
    model = get_model()
    embeddings = model.encode(texts, normalize=True)
    return [e.tolist() for e in embeddings]
```

```python
import uuid
from app.db.qdrant import get_client, ensure_collection

COLLECTION_NAME = "grain_knowledge"

def store_vectors(kb_id: int, title: str, chunks: list, source: str, source_type: str, publish_time: str = None):
    """存储向量到 Qdrant"""
    client = get_client()
    ensure_collection()

    vector_ids = []
    points = []

    for i, chunk in enumerate(chunks):
        from app.services.embed import embed_text
        vector = embed_text(chunk)

        point_id = str(uuid.uuid4())
        vector_ids.append(point_id)

        payload = {
            "kb_id": kb_id,
            "title": title,
            "content": chunk,
            "source": source,
            "source_type": source_type,
            "publish_time": publish_time,
            "chunk_index": i
        }

        points.append({
            "id": point_id,
            "vector": vector,
            "payload": payload
        })

    # 批量插入
    client.upsert(
        collection_name=COLLECTION_NAME,
        points=points
    )

    return ",".join(vector_ids)

    results = client.search(
        collection_name=COLLECTION_NAME,

    return [
```

```python
router = APIRouter(prefix="/api/knowledge", tags=["knowledge"])

def compute_hash(content: str) -> str:
    return hashlib.sha256(content.encode('utf-8')).hexdigest()

@router.post("/ingest")
async def ingest_reports(req: IngestRequest):
    """接收采集系统推送的报告数据"""
    results = []
    ingested = 0
    failed = 0

    for report in req.reports:
        try:
            content_hash = compute_hash(report.content)

            # 检查是否重复
            existing = execute_query(
                "SELECT id FROM t_knowledge_base WHERE content_hash = %s AND deleted = 0",
                (content_hash,)
            )
            if existing:
                results.append({"title": report.title, "kbId": existing[0]['id'], "status": "duplicate"})
                ingested += 1
                continue

            # 插入 MySQL
            sql = """
                INSERT INTO t_knowledge_base
                (title, source_type, source_name, original_url, author, publish_time, content, content_hash, execution_id, vector_status)
                VALUES (%s, 'collection', %s, %s, %s, %s, %s, %s, %s, 'pending')
            """
            kb_id = execute_update(sql, (
                report.title, report.source, report.url, report.author,
                report.publishTime, report.content, content_hash, req.executionId
            ))

            # 触发异步向量化
            # (简化处理，实际应该用 asyncio.to_thread 或 celery)
            try:
                chunks = chunk_text(clean_text(report.content))
                if chunks:
                    vector_ids = store_vectors(
                        kb_id=kb_id,
                        title=report.title,
                        chunks=chunks,
                        source=report.source,
                        source_type="collection",
                        publish_time=report.publishTime
                    )
                    execute_update(
                        "UPDATE t_knowledge_base SET vector_status = 'completed', chunk_count = %s, vector_ids = %s WHERE id = %s",
                        (len(chunks), vector_ids, kb_id)
                    )
                else:
                    execute_update("UPDATE t_knowledge_base SET vector_status = 'failed' WHERE id = %s", (kb_id,))
            except Exception as e:
                execute_update("UPDATE t_knowledge_base SET vector_status = 'failed' WHERE id = %s", (kb_id,))

            results.append({"title": report.title, "kbId": kb_id, "status": "completed"})
            ingested += 1

        except Exception as e:
            results.append({"title": report.title, "kbId": 0, "status": "failed", "error": str(e)})
            failed += 1

    return {"code": 200, "data": {"executionId": req.executionId, "ingested": ingested, "failed": failed, "results": results}}

@router.post("/upload")
async def upload_document(
    file: UploadFile = File(...),
    title: Optional[str] = Form(None),
    source: Optional[str] = Form(None),

    """上传文档并自动处理"""
    # 保存文件到临时目录

    """重新向量化"""
    items = execute_query("SELECT * FROM t_knowledge_base WHERE id = %s", (kb_id,))
    if not items:
        raise HTTPException(status_code=404, detail="知识不存在")

    item = items[0]
    execute_update("UPDATE t_knowledge_base SET vector_status = 'processing' WHERE id = %s", (kb_id,))

    try:
        # 删除旧向量
        if item['vector_ids']:
            delete_vectors(item['vector_ids'])

        chunks = chunk_text(clean_text(item['content']))
        if chunks:
            vector_ids = store_vectors(kb_id, item['title'], chunks, item['source_name'], item['source_type'], str(item['publish_time']) if item['publish_time'] else None)
            execute_update("UPDATE t_knowledge_base SET vector_status = 'completed', chunk_count = %s, vector_ids = %s WHERE id = %s",
                (len(chunks), vector_ids, kb_id))
        else:
            execute_update("UPDATE t_knowledge_base SET vector_status = 'completed', chunk_count = 0 WHERE id = %s", (kb_id,))

        return {"code": 200, "data": {"message": "重新向量化完成"}}
    except Exception as e:
        execute_update("UPDATE t_knowledge_base SET vector_status = 'failed' WHERE id = %s", (kb_id,))
```

```java
private void triggerKnowledgeIngest(String executionId) {
    // 查询该执行的所有报告
    List<Report> reports = reportRepository.findByExecutionId(executionId);
    if (reports.isEmpty()) {
        return;
    }

    String url = "http://localhost:5002/api/knowledge/ingest";
    Map<String, Object> payload = new HashMap<>();
    payload.put("executionId", executionId);
    payload.put("source", "liangxinwang");
    payload.put("reports", reports.stream().map(r -> {
        Map<String, Object> item = new HashMap<>();
        item.put("title", r.getTitle());
        item.put("source", r.getSource());
        item.put("url", r.getOriginalUrl());
        item.put("author", r.getAuthor());
        item.put("publishTime", r.getPublishTime() != null ? r.getPublishTime().toString() : null);
        item.put("content", r.getContent());
        return item;
    }).collect(Collectors.toList()));

    // 重试 3 次
    int retryCount = 0;
    while (retryCount < 3) {
        try {
            restTemplate.postForObject(url, payload, Map.class);
            log.info("知识库接入成功: executionId={}", executionId);
            return;
        } catch (Exception e) {
            retryCount++;
            log.warn("知识库接入重试 {}: executionId={}", retryCount, executionId);
            if (retryCount >= 3) {
                log.error("知识库接入失败，已重试3次: executionId={}", executionId);
            } else {
                Thread.sleep(1000 * retryCount);
            }
        }
    }
```
